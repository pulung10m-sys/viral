import { db } from "./firebase.js"; import { collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/12.0.0/firebase-firestore.js"; window.simpanCampaign = async function(){ let judul = document.getElementById("judul").value; let reward = document.getElementById("reward").value; let kategori = document.getElementById("kategori").value; let deskripsi = document.getElementById("deskripsi").value; try{ await addDoc( collection(db,"campaigns"), { judul:judul, reward:reward, kategori:kategori, deskripsi:deskripsi, status:"Aktif", createdAt:serverTimestamp() } ); alert("Campaign berhasil disimpan"); }catch(error){ alert(error.message); } }
